import hashlib

#Define a salted password
string = "saltedpassword" #Hashlib and variable saltedpassword already created

#Set number of iterations
iters = 3

#Reproduce hash multiple times
for i in range(0, iters):
 md5_hash = hashlib.md5(string.encode()).hexdigest()
sha256_hash = hashlib.sha256(string.encode()).hexdigest()

print('Hash ' + str(i+1) + ' of ' + string)
print('MD5: ' + md5_hash)
print('SHA256: ' + sha256_hash)
