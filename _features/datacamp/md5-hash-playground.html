#Import the hashlib package import hashlib #Define a salted password string = "saltedpassword" #Hash the password using an MD5 algorithm md5_hash = hashlib.md5(string.encode()).hexdigest() sha256_hash = hashlib.sha256(string.encode()).hexdigest() #Print results of hash functions print('Two hashes of ' + string) print('MD5: ' + md5_hash) print('SHA256: ' + sha256_hash)
